 17    def __init__(
 18        self,
 19        deepcopy_on_save=True,
 20        deepcopy_on_retrieve=True,
 21        query_class=None,
 22        max_snapshots=None,
 23        cmp_function=None,
 24    ):
 25        """
 26        Initialize the SnapshotManager.
 27
 28        Args:
 29            deepcopy_on_save (bool): Whether to deepcopy PyTrees when saving snapshots. Defaults to True.
 30            deepcopy_on_retrieve (bool): Whether to return deep copies of PyTrees when retrieving snapshots. Defaults to True.
 31            query_class (type, optional): A class implementing SnapshotQueryInterface. Defaults to SnapshotQuery.
 32            max_snapshots (int, optional): Maximum number of snapshots to store. Defaults to None (no limit).
 33            cmp_function (callable, optional): A comparison function to order snapshots, also used to decide which snapshot to remove
 34                                            when the storage limit is reached. Defaults to None.
 35        """
 36        if query_class and not issubclass(query_class, SnapshotQueryInterface):
 37            raise TypeError("query_class must implement SnapshotQueryInterface.")
 38
 39        self.storage = SnapshotStorage(
 40            max_snapshots=max_snapshots, cmp_function=cmp_function
 41        )
 42        self.query = (query_class or SnapshotQuery)(self.storage.snapshots)
 43        self.deepcopy_on_save = deepcopy_on_save
 44        self.deepcopy_on_retrieve = deepcopy_on_retrieve

 46    def __getitem__(self, index, deepcopy=DEFAULT):
 47        """
 48        Retrieve a Snapshot by index or ID.
 49
 50        Args:
 51            index (int or str):
 52                - If an integer, retrieves the Snapshot by its position in the order of creation.
 53                - If a string, retrieves the Snapshot by its ID.
 54            deepcopy (bool, optional): Whether to return a deep copy of the Snapshot's data. Defaults to the manager's deepcopy setting.
 55
 56        Returns:
 57            The data of the Snapshot.
 58
 59        Raises:
 60            ValueError: If the index or ID is invalid.
 61        """
 62        if isinstance(index, int):
 63            # Get snapshot ID by index using the storage layer
 64            try:
 65                snapshot_id = self.storage.get_snapshot_id_by_index(index)
 66            except IndexError:
 67                raise ValueError(f"Index '{index}' is out of range.")
 68        elif isinstance(index, str):
 69            # Verify the ID exists
 70            if not self.storage.has_snapshot(index):
 71                raise ValueError(f"Snapshot ID '{index}' does not exist.")
 72            snapshot_id = index
 73        else:
 74            raise ValueError(f"Invalid index type: {type(index)}. Must be int or str.")
 75
 76        # Retrieve the snapshot and return its data
 77        snapshot = self.storage.get_snapshot(snapshot_id)
 78
 79        return snapshot.get_data(self._should_deepcopy_on_retrieve(deepcopy))

 83    def save_snapshot(
 84        self,
 85        data,
 86        snapshot_id=None,
 87        metadata=None,
 88        tags=None,
 89        overwrite=False,
 90        deepcopy=DEFAULT,
 91    ):
 92        """
 93        Save a new Snapshot or overwrite an existing one.
 94
 95        Args:
 96            data: The data to store in the Snapshot.
 97            snapshot_id (str, optional): Identifier for the Snapshot. A unique ID is generated if not provided.
 98            metadata (dict, optional): Metadata to associate with the snapshot.
 99            tags (list, optional): Tags to associate with the snapshot.
100            overwrite (bool): Whether to overwrite an existing snapshot.
101            deepcopy (bool, optional): Whether to override the default deepcopy_on_save setting.
102
103
104        Returns:
105            str: The ID of the saved snapshot.
106        """
107        snapshot_id = snapshot_id or str(uuid.uuid4())
108
109        deepcopy = deepcopy if deepcopy is not DEFAULT else self.deepcopy_on_save
110
111        snapshot = Snapshot(data, metadata, tags, deepcopy=deepcopy)
112        self.storage.add_snapshot(snapshot_id, snapshot, overwrite=overwrite)
113        return snapshot_id

179        """
180        Clone a snapshot with an optional update to metadata.

183            snapshot_id (str): The ID of the snapshot to clone.
184            metadata (dict, optional): Metadata to add or update in the cloned snapshot.

186        Returns:
187            str: The ID of the cloned snapshot.
188
189        Raises:
190            ValueError: If the snapshot ID does not exist.
191        """

196
197        # Merge metadata (if provided) with the original snapshot's metadata
198        cloned_metadata = original_snapshot.metadata.copy()
199        if metadata:
200            cloned_metadata.update(metadata)
201
202        # Save the cloned snapshot with a new ID
203        cloned_snapshot_id = str(uuid.uuid4())
204        self.save_snapshot(
205            cloned_pytree, snapshot_id=cloned_snapshot_id, metadata=cloned_metadata
206        )
207
208        return cloned_snapshot_id

419    @staticmethod
420    def load_state(file_path):
421        """
422        Load a SnapshotManager state from a file.
423
424        Args:
425            file_path (str): Path to the file containing the saved state.
426
427        Returns:
428            SnapshotManager: A new manager instance with the loaded state.
429        """
430        state = SnapshotPersistence.load_state(file_path)
431
432        # Create a new manager with the loaded state
433        manager = SnapshotManager(
434            max_snapshots=state["max_snapshots"],
435            deepcopy_on_save=state["deepcopy_on_save"],
436            deepcopy_on_retrieve=state["deepcopy_on_retrieve"],
437        )
438
439        # Restore snapshots into the manager's storage
440        for snapshot_id, snapshot_data in state["snapshots"].items():
441            snapshot = Snapshot.from_dict(snapshot_data)
442            manager.storage.add_snapshot(snapshot_id, snapshot)
443
444        return manager

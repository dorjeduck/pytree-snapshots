 65    def save_snapshot(
 66        self,
 67        pytree,
 68        snapshot_id=None,
 69        metadata=None,
 70        tags=None,
 71        compress=False,
 72        overwrite=False,
 73    ):
 74        """
 75        Save a new PytreeSnapshot or overwrite an existing one.
 76
 77        Args:
 78            pytree: The PyTree to store in the PytreeSnapshot.
 79            snapshot_id (str, optional): Identifier for the PytreeSnapshot. A unique ID is generated if not provided.
 80            metadata (dict, optional): Additional metadata to associate with the PytreeSnapshot. Defaults to an empty dictionary.
 81            tags (list, optional): Tags to associate with the PytreeSnapshot for organization. Defaults to an empty list.
 82            compress (bool): Whether to compress the PytreeSnapshot. Defaults to False.
 83            overwrite (bool): Whether to overwrite an existing PytreeSnapshot with the same ID. Defaults to False.
 84
 85        Returns:
 86            str: The ID of the saved PytreeSnapshot.
 87
 88        Raises:
 89            ValueError: If the PytreeSnapshot ID already exists and `overwrite` is False.
 90
 91        Examples:
 92            Save a new PytreeSnapshot:
 93                snapshot_id = manager.save_snapshot(pytree, metadata={"project": "test"}, tags=["experiment"])
 94
 95            Overwrite an existing PytreeSnapshot:
 96                manager.save_snapshot(pytree, snapshot_id="existing_id", overwrite=True)
 97        """
 98        snapshot_id = snapshot_id or str(uuid.uuid4())
 99
100        if snapshot_id in self.snapshots and not overwrite:
101            raise ValueError(
102                f"PytreeSnapshot ID '{snapshot_id}' already exists. Use overwrite=True to update it."
103            )
104
105        if snapshot_id not in self.snapshots:
106            # Enforce max_snapshots limit
107            if (
108                self.max_snapshots is not None
109                and len(self.snapshots) >= self.max_snapshots
110            ):
111                oldest_snapshot = self.snapshot_order[0]
112                self._remove_snapshot(oldest_snapshot)
113
114            self.snapshot_order.append(snapshot_id)
115
116        # Create and store the snapshot
117        self.snapshots[snapshot_id] = PytreeSnapshot(pytree, metadata, tags, compress)
118        return snapshot_id

180    def clone_snapshot(self, snapshot_id, new_snapshot_id=None):
181        """
182        Clone an existing PytreeSnapshot with a new ID.
183
184        Args:
185            snapshot_id (str): The ID of the PytreeSnapshot to clone.
186            new_snapshot_id (str, optional): The ID to assign to the cloned PytreeSnapshot. A unique ID is generated if not provided.
187
188        Returns:
189            str: The ID of the cloned PytreeSnapshot.
190
191        Raises:
192            ValueError: If the PytreeSnapshot ID to clone does not exist or the new PytreeSnapshot ID already exists.
193
194        Examples:
195            Clone a PytreeSnapshot with a new ID:
196                cloned_id = manager.clone_snapshot("existing_id")
197        """
198        if snapshot_id not in self.snapshots:
199            raise ValueError(f"PytreeSnapshot ID '{snapshot_id}' does not exist.")
200
201        # Generate a new snapshot ID if none is provided
202        new_snapshot_id = new_snapshot_id or str(uuid.uuid4())
203
204        if new_snapshot_id in self.snapshots:
205            raise ValueError(f"PytreeSnapshot ID '{new_snapshot_id}' already exists.")
206
207        # Clone the snapshot
208        original_snapshot = self.snapshots[snapshot_id]
209        pytree = original_snapshot.get_pytree(deepcopy=True)
210        metadata = original_snapshot.metadata.copy()
211        tags = original_snapshot.tags.copy()
212        compress = original_snapshot.compress
213
214        # Save the cloned snapshot
215        self.snapshots[new_snapshot_id] = PytreeSnapshot(
216            pytree=pytree, metadata=metadata, tags=tags, compress=compress
217        )
218        self.snapshot_order.append(new_snapshot_id)
219
220        return new_snapshot_id

460    def compare_snapshots(
461        self,
462        snapshot_id1,
463        snapshot_id2,
464        custom_comparator=None,
465        condition=None,
466    ):
467        """
468        Compare two PytreeSnapshots and return their differences as a PyTree.
469
470        Args:
471            snapshot_id1 (str): The ID of the first PytreeSnapshot.
472            snapshot_id2 (str): The ID of the second PytreeSnapshot.
473            custom_comparator (callable, optional): A custom function to compare leaves. Defaults to None.
474                - The function should take two arguments (leaf1, leaf2) and return:
475                    - None: If the leaves are considered equal.
476                    - A custom difference representation otherwise.
477            condition (callable, optional): A function that takes a leaf value and returns True if it should be compared, False otherwise.
478                - If the condition is False, the leaf is excluded from the comparison.
479            return_paths (bool): If True, include paths to differing elements in the result. Defaults to True.
480
481        Returns:
482            dict: Differences between the two PytreeSnapshots.
483                - Keys are paths (if `return_paths=True`) or an index-like representation.
484                - Values are tuples: (value_in_snapshot1, value_in_snapshot2) or the custom comparison result.
485
486        Raises:
487            ValueError: If either PytreeSnapshot ID does not exist.
488
489        Examples:
490            Save two PytreeSnapshots with different PyTrees:
491            >>> pytree1 = {"a": jnp.array([1, 2, 3]), "b": {"x": 42, "y": "hello"}}
492            >>> pytree2 = {"a": jnp.array([1, 2, 4]), "b": {"x": 42, "y": "world"}}
493            >>> manager.save_snapshot(pytree1, snapshot_id="id1")
494            >>> manager.save_snapshot(pytree2, snapshot_id="id2")
495
496            Compare the two PytreeSnapshots:
497            >>> differences = manager.compare_snapshots(
498                    "id1", "id2",
499                    condition=lambda x: isinstance(x, jnp.ndarray)
500                )
501            >>> print(differences)
502
503            Output:
504            {
505                "a": (Array([1, 2, 3], dtype=int32), Array([1, 2, 4], dtype=int32))
506            }
507        """
508        if snapshot_id1 not in self.snapshots or snapshot_id2 not in self.snapshots:
509            raise ValueError("Both PytreeSnapshot IDs must exist.")
510
511        pytree1 = self.snapshots[snapshot_id1].get_pytree(deepcopy=False)
512        pytree2 = self.snapshots[snapshot_id2].get_pytree(deepcopy=False)
513
514        # Unify structures
515        pytree1, pytree2 = self.unify_pytree_structures(pytree1, pytree2)
516
517        def default_comparator(x, y):
518            if isinstance(x, jnp.ndarray) and isinstance(y, jnp.ndarray):
519                return None if jnp.array_equal(x, y) else (x, y)
520            if isinstance(x, np.ndarray) and isinstance(y, np.ndarray):
521                return None if np.array_equal(x, y) else (x, y)
522            if isinstance(x, (int, float, str, bool)) and isinstance(
523                y, (int, float, str, bool)
524            ):
525                return (x, y) if x != y else None
526            if type(x) != type(y) or x != y:
527                return (x, y)
528            return None
529
530        comparator = (
531            (lambda x, y: None if custom_comparator(x, y) else (x, y))
532            if custom_comparator
533            else default_comparator
534        )
535
536        def collect_differences(x, y):
537            # Apply condition
538            if condition and not (condition(x) and condition(y)):
539                return Ellipsis  # Represent excluded elements with Ellipsis
540            result = comparator(x, y)
541            return result if result is not None else None
542
543        differences = jax.tree.map(collect_differences, pytree1, pytree2)
544
545        return differences

547    @staticmethod
548    def unify_pytree_structures(pytree1, pytree2, placeholder=Ellipsis):
549        """
550        Align the structures of two PyTrees, filling in missing keys or elements with a placeholder.
551
552        Args:
553            pytree1: The first PyTree (can be a nested dict/list/tuple).
554            pytree2: The second PyTree (can be a nested dict/list/tuple).
555            placeholder: The value to use for missing elements. Defaults to Ellipsis.
556
557        Returns:
558            Tuple: Two PyTrees with unified structure.
559        """
560        if isinstance(pytree1, dict) and isinstance(pytree2, dict):
561            # Union of all keys
562            all_keys = set(pytree1.keys()).union(pytree2.keys())
563            aligned1 = {}
564            aligned2 = {}
565            for key in all_keys:
566                aligned1[key], aligned2[key] = PytreeSnapshotManager.unify_pytree_structures(
567                    pytree1.get(key, placeholder),
568                    pytree2.get(key, placeholder),
569                    placeholder,
570                )
571            return aligned1, aligned2
572        elif isinstance(pytree1, (list, tuple)) and isinstance(pytree2, (list, tuple)):
573            # Align lists/tuples
574            max_len = max(len(pytree1), len(pytree2))
575            aligned1 = list(pytree1) + [placeholder] * (max_len - len(pytree1))
576            aligned2 = list(pytree2) + [placeholder] * (max_len - len(pytree2))
577            if isinstance(pytree1, tuple):
578                aligned1 = tuple(aligned1)
579            if isinstance(pytree2, tuple):
580                aligned2 = tuple(aligned2)
581            return aligned1, aligned2
582        else:
583            # For leaves or mismatched types, return as-is
584            return pytree1, pytree2

623    @staticmethod
624    def load_state(file_path, decompress=False):
625        """
626        Load the PytreeSnapshotManager state from a file.
627
628        Args:
629            file_path (str): Path to the file containing the saved state.
630            decompress (bool): Whether the state file is compressed. Defaults to False.
631
632        Returns:
633            PytreeSnapshotManager: An instance of PytreeSnapshotManager with the loaded state.
634
635        Raises:
636            ValueError: If decompression or deserialization fails.
637            FileNotFoundError: If the specified file does not exist.
638
639        Examples:
640            Load the manager's state:
641                manager = PytreeSnapshotManager.load_state("state.pkl", decompress=True)
642        """
643        try:
644            # Read the file
645            with open(file_path, "rb") as file:
646                serialized_data = file.read()
647
648            # Decompress if needed
649            if decompress:
650                try:
651                    serialized_data = zlib.decompress(serialized_data)
652                except zlib.error as e:
653                    raise ValueError(f"Decompression failed: {e}")
654
655            # Deserialize the state
656            state = pickle.loads(serialized_data)
657        except FileNotFoundError:
658            raise FileNotFoundError(f"The file '{file_path}' does not exist.")
659        except (pickle.PickleError, EOFError) as e:
660            raise ValueError(f"Deserialization failed: {e}")
661
662        # Recreate PytreeSnapshotManager instance
663        manager = PytreeSnapshotManager(
664            max_snapshots=state["max_snapshots"],
665            deepcopy=state["deepcopy"],
666        )
667
668        # Convert dictionaries back to PytreeSnapshot objects
669        try:

461    def compare_snapshots(
462        self,
463        snapshot_id1,
464        snapshot_id2,
465        custom_comparator=None,
466        condition=None,
467    ):
468        """
469        Compare two PytreeSnapshots and return their differences as a PyTree.
470
471        Args:
472            snapshot_id1 (str): The ID of the first PytreeSnapshot.
473            snapshot_id2 (str): The ID of the second PytreeSnapshot.
474            custom_comparator (callable, optional): A custom function to compare leaves. Defaults to None.
475                - The function should take two arguments (leaf1, leaf2) and return:
476                    - None: If the leaves are considered equal.
477                    - A custom difference representation otherwise.
478            condition (callable, optional): A function that takes a leaf value and returns True if it should be compared, False otherwise.
479                - If the condition is False, the leaf is excluded from the comparison.
480            return_paths (bool): If True, include paths to differing elements in the result. Defaults to True.
481
482        Returns:
483            dict: Differences between the two PytreeSnapshots.
484                - Keys are paths (if `return_paths=True`) or an index-like representation.
485                - Values are tuples: (value_in_snapshot1, value_in_snapshot2) or the custom comparison result.
486
487        Raises:
488            ValueError: If either PytreeSnapshot ID does not exist.
489
490        Examples:
491            Save two PytreeSnapshots with different PyTrees:
492            >>> pytree1 = {"a": jnp.array([1, 2, 3]), "b": {"x": 42, "y": "hello"}}
493            >>> pytree2 = {"a": jnp.array([1, 2, 4]), "b": {"x": 42, "y": "world"}}
494            >>> manager.save_snapshot(pytree1, snapshot_id="id1")
495            >>> manager.save_snapshot(pytree2, snapshot_id="id2")
496
497            Compare the two PytreeSnapshots:
498            >>> differences = manager.compare_snapshots(
499                    "id1", "id2",
500                    condition=lambda x: isinstance(x, jnp.ndarray)
501                )
502            >>> print(differences)
503
504            Output:
505            {
506                "a": (Array([1, 2, 3], dtype=int32), Array([1, 2, 4], dtype=int32))
507            }
508        """
509        if snapshot_id1 not in self.snapshots or snapshot_id2 not in self.snapshots:
510            raise ValueError("Both PytreeSnapshot IDs must exist.")
511
512        pytree1 = self.snapshots[snapshot_id1].get_pytree(deepcopy=False)
513        pytree2 = self.snapshots[snapshot_id2].get_pytree(deepcopy=False)
514
515        # Unify structures
516        pytree1, pytree2 = self.unify_pytree_structures(pytree1, pytree2)
517
518        def default_comparator(x, y):
519            if isinstance(x, jnp.ndarray) and isinstance(y, jnp.ndarray):
520                return None if jnp.array_equal(x, y) else (x, y)
521            if isinstance(x, np.ndarray) and isinstance(y, np.ndarray):
522                return None if np.array_equal(x, y) else (x, y)
523            if isinstance(x, (int, float, str, bool)) and isinstance(
524                y, (int, float, str, bool)
525            ):
526                return (x, y) if x != y else None
527            if type(x) != type(y) or x != y:
528                return (x, y)
529            return None
530
531        comparator = (
532            (lambda x, y: None if custom_comparator(x, y) else (x, y))
533            if custom_comparator
534            else default_comparator
535        )
536
537        def collect_differences(x, y):
538            # Apply condition
539            if condition and not (condition(x) and condition(y)):
540                return Ellipsis  # Represent excluded elements with Ellipsis
541            result = comparator(x, y)
542            return result if result is not None else None
543
544        differences = jax.tree.map(collect_differences, pytree1, pytree2)
545
546        return differences

548    @staticmethod
549    def unify_pytree_structures(pytree1, pytree2, placeholder=Ellipsis):
550        """
551        Align the structures of two PyTrees, filling in missing keys or elements with a placeholder.
552
553        Args:
554            pytree1: The first PyTree (can be a nested dict/list/tuple).
555            pytree2: The second PyTree (can be a nested dict/list/tuple).
556            placeholder: The value to use for missing elements. Defaults to Ellipsis.
557
558        Returns:
559            Tuple: Two PyTrees with unified structure.
560        """
561        if isinstance(pytree1, dict) and isinstance(pytree2, dict):
562            # Union of all keys
563            all_keys = set(pytree1.keys()).union(pytree2.keys())
564            aligned1 = {}
565            aligned2 = {}
566            for key in all_keys:
567                aligned1[key], aligned2[key] = PytreeSnapshotManager.unify_pytree_structures(
568                    pytree1.get(key, placeholder),
569                    pytree2.get(key, placeholder),
570                    placeholder,
571                )
572            return aligned1, aligned2
573        elif isinstance(pytree1, (list, tuple)) and isinstance(pytree2, (list, tuple)):
574            # Align lists/tuples
575            max_len = max(len(pytree1), len(pytree2))
576            aligned1 = list(pytree1) + [placeholder] * (max_len - len(pytree1))
577            aligned2 = list(pytree2) + [placeholder] * (max_len - len(pytree2))
578            if isinstance(pytree1, tuple):
579                aligned1 = tuple(aligned1)
580            if isinstance(pytree2, tuple):
581                aligned2 = tuple(aligned2)
582            return aligned1, aligned2
583        else:
584            # For leaves or mismatched types, return as-is
585            return pytree1, pytree2
